% -----------------------------------------------------------------------------
% Layout patch: the upstream russell.cls definition of \cventrynew uses a
% negative \vspace{-1.5mm} at the top of every entry and a vertically-centered
% L{} column for the description. With multi-bullet entries (cvitems contains
% a -4mm \vspace at its start), LaTeX under-estimates the cell height and the
% next entry's title row visually collides with the previous entry's last
% bullets. We override \cventrynew here to remove the negative leading vspace,
% add a small trailing padding inside the description cell, and put a fixed
% gutter between consecutive entries. The visible layout is otherwise
% identical to russell.cls's original.
% -----------------------------------------------------------------------------
% NB: this is intentionally \renewcommand (not \renewcommand*) so the
% description body (#3) can legitimately contain \par / blank lines
% (e.g. a small italic company-context line followed by the cvitems list).
% We inline \descriptionstyle's formatting instead of passing #3 through it,
% because \descriptionstyle in russell.cls is itself a starred command and
% would reject \par inside its argument.
\renewcommand{\cventrynew}[3]{%
  \setlength\tabcolsep{0pt}%
  \setlength{\extrarowheight}{0pt}%
  \begin{tabular*}{\textwidth}{@{\extracolsep{\fill}} L{\textwidth - 4.5cm} R{4.5cm}}
    \ifempty{#1#2}
      {\entrydatestyle{#2} \\}
      {\entrytitlestyle{#1} & \entrylocationstyle{#2} \\}
    \multicolumn{2}{L{\textwidth}}{%
      {\fontsize{10pt}{1em}\bodyfontlight\upshape\color{text}%
        #3%
      }\vspace{1mm}%
    }
  \end{tabular*}%
  \par\addvspace{3mm}%
}
